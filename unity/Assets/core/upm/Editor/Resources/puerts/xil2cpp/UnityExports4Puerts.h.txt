struct WrapData 
{
    WrapFuncPtr Wrap;
    MethodType* Method;
    MethodPointer MethodPointer;
    bool IsStatic;
    bool IsExtensionMethod;
    bool HasParamArray;
    int OptionalNum;
    TypeIdType* TypeInfos[0];
};

{
    const void* TypeId;
    const void* SuperTypeId;
    TypeIdType* Class;

};
